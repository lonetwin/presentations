    \section{Introduction}
        \subsection{What is Free and Open Source Software ?}
        \frame {
            \frametitle{What is Free and Open Source Software ?}
            \begin{itemize}[<+->]
            \item Free as in mukt not muft
                \begin{itemize}
                \item FOSS != Freeware
                \item FOSS != Shared Source
                \item FOSS != Public Domain
                \end{itemize}
            \item Not a new concept
                \begin{itemize}
                \item Open research, Open standards, Open source !
                \end{itemize}
            \end{itemize}
        }

        \subsection{Why should you care ?}
        \frame {
            \frametitle{Why should you care ?}
            \begin{itemize}[<+->]
            \item FOSS has influenced the software industry
            \item FOSS will continue to influence the software industry
            \item FOSS concepts have influenced and will continue to influence other industries
            \item FOSS makes you smarter :)
            \end{itemize}
        }

        \subsection{What is the FOSS ecosystem ?}
        \frame {
            \frametitle{What is the FOSS ecosystem ?}
            \begin{itemize}[<+->]
            \item The properties:
                \begin{itemize}
                \item Evolutionary
                \item Self-regulating
                \item Fault-tolerant
                \item So, FOSS ecosystem is an 'Intelligent positive feedback system'
                \end{itemize}
            \item The participants:
                \begin{itemize}
                \item No clear distinction of participants \\
                consumers are producers and vice versa
                \item Contribution is not just technical. Translators, \\
                content creators, designers, in fact anyone who can add \\
                value can participate.
                \end{itemize}
            \end{itemize}
        }
        \frame {
            \frametitle{How is the FOSS ecosystem different ?}
            \begin{itemize}
            \item<1->FOSS versus proprietary software ecosystems
                \begin{itemize}
                \item<1->Driven by needs instead of 'market size'
                \item<1->Feedback loop is more responsive and more reinforcing
                \item<1->Longer lifetime
                \end{itemize}
            \item<2->FOSS ecosystem versus an online community \\
            Users are contributors not just members
            \end{itemize}
        }

    \section{Getting involved}
        \subsection{As a user}
        \frame {
            \frametitle{Becoming a FOSS user}
            Install and use FOSS
            \begin{itemize}[<+->]
            \item FOSS alternatives on Windows

            Firefox, VLC, Pidgin, OpenOffice, PDFCreator, ClamWin ...
            \item Linux Distributions
                \begin{itemize}
                \item What is a distro ?
                \item Which distro should I use ? Fedora ? Ubuntu ? ...
                \item Dedicated flavors - Edubuntu, Fedora Electronic lab, Ubuntu studio ...
                \item Explore and learn your tools -- think why ?
                \end{itemize}
            \end{itemize}
        }
        \subsection{As a community member}
        \frame {
            \frametitle{As a community member}
            \begin{itemize}
            \item<1->Read FOSS related news, blogs ...etc
            \item<2->Join mailing lists, LUGs
            \item<3->Ask questions, give answers -- learn and share as much as
            possible in your free time
            \item<4->...but remember -- everyone is also doing the same ! so,
                \begin{itemize}
                \item<5->Respect the rules (for example: mailing list rules, irc
            conventions, etc)
                \item<5->Ask smart questions
                \item<5->Acknowledge the value of the community
                \item<6->Be kind to the n00bs
                \end{itemize}
            \end{itemize}
        }
        \subsection{As a contributor}
        \frame {
            \frametitle{As a contributor}
            \begin{itemize}[<+->]
            \item Developer, content writer, designer, translator, all of them ?
            \item Submit bug reports and fix bugs
            \item Become a package maintainer
            \item Scratch your own itch
            \item Innovate, Share, Repeat !
            \end{itemize}
        }
